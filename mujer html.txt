<script>

const poema = document.getElementById("poema");
const editor = document.getElementById("editor");

function guardar(){
    localStorage.setItem("mensaje", editor.value);
    poema.innerText = editor.value;
}

window.onload = function(){
    const guardado = localStorage.getItem("mensaje");
    if(guardado){
        poema.innerText = guardado;
        editor.value = guardado;
    }
}
